─────────────────────────────
 1️⃣ Basic Variable Expansion
─────────────────────────────
echo $HOME          # Expected: <your $HOME value>
echo "$HOME"        # Expected: <your $HOME value>
echo '$HOME'        # Expected: $HOME
echo $USER          # Expected: <your $USER value>
echo "$USER"        # Expected: <your $USER value>
echo '$USER'        # Expected: $USER


─────────────────────────────
 2️⃣ Mixed Quoting
─────────────────────────────
echo "'$HOME'"      # Expected: '<your $HOME value>'
echo '"$HOME"'      # Expected: "$HOME"
echo "'$HOME'"      # Expected: '<your $HOME value>'
echo "\"$HOME\""    # Expected: "<your $HOME value>"
echo "'$USER'"      # Expected: '<your $USER value>'
echo '"$USER"'      # Expected: "$USER"


─────────────────────────────
 3️⃣ Edge Cases
─────────────────────────────
echo ''             # Expected: (empty line)
echo ""             # Expected: (empty line)
echo $UNDEFINED_VAR # Expected: (empty line)
echo "$UNDEFINED_VAR" # Expected: (empty line)
echo '$UNDEFINED_VAR' # Expected: $UNDEFINED_VAR


─────────────────────────────
 4️⃣ Multiple Variables
─────────────────────────────
echo $HOME$USER     # Expected: <your $HOME value><your $USER value>
echo "$HOME$USER"   # Expected: <your $HOME value><your $USER value>
echo '$HOME$USER'   # Expected: $HOME$USER


─────────────────────────────
 5️⃣ Spaces and Tabs
─────────────────────────────
echo    $HOME   $USER      # Expected: <your $HOME value> <your $USER value>
echo "$HOME   $USER"       # Expected: <your $HOME value>   <your $USER value>
echo '$HOME   $USER'       # Expected: $HOME   $USER


─────────────────────────────
 6️⃣ Nested Quotes
─────────────────────────────
echo "'\"$HOME\"'"     # Expected: '"<your $HOME value>"'
echo '"\'$HOME\'"'     # Expected: "'$HOME'"


─────────────────────────────
 7️⃣ Redirections and Pipes
─────────────────────────────
echo $HOME > out.txt    # Expected: (output written to out.txt: <your $HOME value>)
echo "$HOME" | cat      # Expected: <your $HOME value>


─────────────────────────────
 8️⃣ Unclosed Quotes
─────────────────────────────
echo "unclosed          # Expected: Error: unclosed quote
echo 'unclosed         # Expected: Error: unclosed quote


─────────────────────────────
 9️⃣ Special Characters
─────────────────────────────
echo $HOME!           # Expected: <your $HOME value>!
echo "$HOME!"         # Expected: <your $HOME value>!
echo '$HOME!'         # Expected: $HOME!
